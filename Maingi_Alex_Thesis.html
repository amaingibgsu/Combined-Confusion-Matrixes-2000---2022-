for i, (conf_matrix, overall_accuracy, producer_accuracy, consumer_accuracy, title) in enumerate(conf_matrix_data):
    ax = axs[i//3, i%3]
    sns.set(font_scale=1.2)  # Adjust font size

    # Include total columns and rows
    class_labels_with_totals = ['water', 'forest', 'vegetation', 'urban', 'barren', 'Total']
    conf_matrix_with_totals = np.column_stack([conf_matrix, np.sum(conf_matrix, axis=1)])
    conf_matrix_with_totals = np.row_stack([conf_matrix_with_totals, np.sum(conf_matrix_with_totals, axis=0)])

    sns.heatmap(conf_matrix_with_totals, annot=True, fmt='d', cmap='Blues',
                xticklabels=class_labels_with_totals, yticklabels=class_labels_with_totals, cbar=False, ax=ax)

    ax.tick_params(axis='x', labeltop=True, labelbottom=False)
    ax.set_xlabel('Predicted')
    ax.set_ylabel('Actual')
    ax.set_title(title)  # Set the title for each subplot

plt.tight_layout()
plt.show()
